import ast

def clean_crew(text):

    names = []

    try:

        crew_list = ast.literal_eval(text)

        for item in crew_list:

            # خد الـ Director فقط
            if item['job'] == 'Director':

                names.append(
                    item['name']
                )

    except:
        pass

    return " ".join(names)

movies['director'] = movies[

].apply(clean_crew)

stop_words = set(
    stopwords.words('english')
)

lemmatizer = WordNetLemmatizer()

def preprocess(text):

    text = str(text).lower()

    # remove special chars
    text = re.sub(
        r'[^a-zA-Z0-9 ]',
        ' ',
        text
    )

    # tokenize
    words = text.split()

    # remove stopwords
    words = [
        w for w in words
        if w not in stop_words
    ]

    # lemmatization
    words = [
        lemmatizer.lemmatize(w)
        for w in words
    ]

    return " ".join(words)

def preprocess_title(text):

    text = str(text).lower()

    # remove special chars only
    text = re.sub(
        r'[^a-zA-Z0-9 ]',
        ' ',
        text
    )

    return text

from sklearn.metrics.pairwise import cosine_similarity

def search_movies(
    query,
    title_weight=2.0,
    overview_weight=1.0,
    cast_weight=1.5,
    director_weight=1.5,
    top_k=10
):

    # =========================
    # preprocess query
    # =========================

    clean_query_title = preprocess_title(query)

    clean_query_overview = preprocess(query)

    # =========================
    # title similarity
    # =========================

    title_query_vector = title_vectorizer.transform(
        [clean_query_title]
    )

    title_scores = cosine_similarity(
        title_query_vector,
        title_matrix
    ).flatten()

    # =========================
    # overview similarity
    # =========================

    overview_query_vector = overview_vectorizer.transform(
        [clean_query_overview]
    )

    overview_scores = cosine_similarity(
        overview_query_vector,
        overview_matrix
    ).flatten()

    # =========================
    # cast similarity
    # =========================

    cast_query_vector = cast_vectorizer.transform(
        [clean_query_title]
    )

    cast_scores = cosine_similarity(
        cast_query_vector,
        cast_matrix
    ).flatten()

    # =========================
    # director similarity
    # =========================

    director_query_vector = director_vectorizer.transform(
        [clean_query_title]
    )

    director_scores = cosine_similarity(
        director_query_vector,
        director_matrix
    ).flatten()

    # =========================
    # final weighted score
    # =========================

    final_scores = (
        title_weight * title_scores
        +
        overview_weight * overview_scores
        +
        cast_weight * cast_scores
        +
        director_weight * director_scores
    )

    # =========================
    # top results
    # =========================

    top_indices = final_scores.argsort()[-top_k:][::-1]

    results = movies.iloc[top_indices][[
        'title',
        'genres_clean',
        'main_cast',
        'director',
        'vote_average',
        'popularity'
    ]]

    results['score'] = final_scores[top_indices]

    return results

from rank_bm25 import BM25Okapi

movies['bm25_text'] = (

    movies['clean_cast'] + ' ' +

tokenized_corpus = [
    doc.split()
    for doc in movies['bm25_text']
]

bm25 = BM25Okapi(
    tokenized_corpus
)

def bm25_search(
    query,
    top_k=10
):

    # preprocess query
    clean_query = preprocess(query)

    # tokenize query
    tokenized_query = clean_query.split()

    # BM25 scores
    scores = bm25.get_scores(
        tokenized_query
    )

    # top results
    top_indices = scores.argsort()[-top_k:][::-1]

    results = movies.iloc[top_indices][[
        'title',
        'genres_clean',
        'main_cast',
        'director',
        'vote_average',
        'popularity'
    ]]

    results['score'] = scores[top_indices]
    results = results.sort_values(
    by=['score', 'popularity'],
    ascending=False
)

    return results

def precision_at_k(
    retrieved_titles,
    relevant_titles,
    k=10
):

    retrieved_k = retrieved_titles[:k]

    relevant_found = 0

    for movie in retrieved_k:

        if movie in relevant_titles:

            relevant_found += 1

    return relevant_found / k

for query, relevant_movies in test_queries.items():
